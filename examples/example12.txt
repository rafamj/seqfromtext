/*
Identity request
*/
seq1=openOutPort("seq1")
seq2=openInPort("seq2")
connectOut(seq1,20,0)
connectIn(seq2,20,0)
d1=sysExChannel(seq1)

ID = 0x7F //Disregard ID

d1<</ X( F0 7E <ID>  06 01 f7 ) /

def  ident(buffer,len) {
  printf("%d ID %x manufacturer %hhX ",len, buffer[2], buffer[5])
  printf("model %x %x family member %d %d ", buffer[6],buffer[7],buffer[8],buffer[9])
  printf("software revision %c%c%c%c \n",buffer[10], buffer[11], buffer[12], buffer[13])
}
waitSysEx(seq2, ident)
